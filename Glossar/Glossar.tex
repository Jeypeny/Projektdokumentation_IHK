% Füge hier deine Glossar definitionen hinzu

\makeglossaries

\newglossaryentry{Kontext}{name={Kontext},description={Zusätzliche Informationen, die ein KI-System benötigt, um ein Situation oder Aufgabe korrekt zu verstehen und zu interpretieren}}

\newglossaryentry{Prompt}{name={prompt},description={Eine Anweisung an ein KI-System, die vorgibt, welche Aufgaben es zu erledigen hat und wie es diese lösen soll}}

\newglossaryentry{API}{name={API},description={Aplication Programming Interface, eine Software Schnittstelle zu einer Applikation mit der man dieses an andere Applikationen anbinden kann und so Befehle und Daten an dieses übergeben kann}}

\newglossaryentry{Token}{name={Token},description={Kleinste Einheit, die von einem Sprachmodell verarbeitet werden kann}}

\newglossaryentry{CLI}{name={CLI},description={Command Line Interface, textbasierte Benutzeroberfläche, mit der man direkt mit dem Betriebssystem interagieren kann, indem man Befehle eintippt}}
